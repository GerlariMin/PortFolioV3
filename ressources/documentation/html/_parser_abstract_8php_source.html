475  protected function handleNamespaces(array $stmts) : array {

478  if (null === $style) {

481  } elseif ('brace' === $style) {

483  $afterFirstNamespace = false;

486  $afterFirstNamespace = true;
487  } elseif (!$stmt instanceof Node\Stmt\HaltCompiler
488  && !$stmt instanceof Node\Stmt\Nop
489  && $afterFirstNamespace && !$hasErrored) {
490  $this->emitError(new Error(
491  'No code may exist outside of namespace {}', $stmt->getAttributes()));
492  $hasErrored = true; // Avoid one error for every statement
493  }
494  }
495  return $stmts;
496  } else {
497  // For semicolon namespaces we have to move the statements after a namespace declaration into ->stmts
498  $resultStmts = [];

504  $this->fixupNamespaceAttributes($lastNs);
505  }
506  if ($stmt->stmts === null) {
507  $stmt->stmts = [];
508  $targetStmts =& $stmt->stmts;
509  $resultStmts[] = $stmt;
510  } else {

512  $resultStmts[] = $stmt;

514  }

516  } elseif ($stmt instanceof Node\Stmt\HaltCompiler) {
517  // __halt_compiler() is not moved into the namespace
518  $resultStmts[] = $stmt;
519  } else {
520  $targetStmts[] = $stmt;
521  }
522  }
523  if ($lastNs !== null) {
524  $this->fixupNamespaceAttributes($lastNs);
525  }
526  return $resultStmts;
527  }
