236  {

258  $result = $this->getTestResultByName($this->originalExecutionOrder[$this->testFlushIndex]);
259  } else {

261  // flush result to output stream right away

263  }
264

268  $this->testFlushIndex++;
269  $prevResult = $result;

274  } while ($flushed && $this->testFlushIndex < $this->testIndex);
